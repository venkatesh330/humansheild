```typescript
function scoreCustomSkill(skillName: string, industry: string | null): number {
  const name = skillName.toLowerCase();
  
  // Layer 1: Hard override blocklist — skills that contain tech keywords 
  // but are actually PROTECTIVE in 2026
  const PROTECTIVE_DESPITE_TECH = [
    'ai ethics', 'ai governance', 'ai audit', 'machine learning ethics',
    'algorithm bias', 'responsible ai', 'ai policy', 'tech regulation',
    'data privacy', 'cybersecurity strategy', 'digital transformation leadership',
    'prompt engineering oversight', 'ai risk management'
  ];
  if (PROTECTIVE_DESPITE_TECH.some(p => name.includes(p))) {
    return Math.round(8 + Math.random() * 7); // 8-15 range — highly protective
  }
  
  // Layer 2: Hard override list — skills that sound human but are automatable
  const AUTOMATABLE_DESPITE_HUMAN_SOUND = [
    'data entry', 'form filling', 'scheduling', 'appointment booking',
    'invoice processing', 'expense reporting', 'basic bookkeeping',
    'file organization', 'meeting notes', 'email sorting'
  ];
  if (AUTOMATABLE_DESPITE_HUMAN_SOUND.some(p => name.includes(p))) {
    return Math.round(82 + Math.random() * 10); // 82-92 range
  }
  
  // Layer 3: Existing keyword scoring with improved weights
  const HUMAN_KEYWORDS = [/* existing 31 */];
  const TECH_KEYWORDS = [/* existing 28 */];
  const humanHits = HUMAN_KEYWORDS.filter(k => name.includes(k)).length;
  const techHits = TECH_KEYWORDS.filter(k => name.includes(k)).length;
  const total = humanHits + techHits || 1;
  
  // Base score — use a more conservative range (10-82 instead of 10-86)
  let base = (techHits / total) * 82 + (humanHits / total) * 10;
  
  // Layer 4: Industry context modifier
  const INDUSTRY_RISK_MODIFIERS: Record<string, number> = {
    'finance': 1.12, 'legal': 1.10, 'marketing': 1.08,
    'healthcare': 0.82, 'mental_health': 0.72, 'education': 0.88,
    'social_work': 0.70, 'architecture': 0.85
  };
  const industryKey = industry?.toLowerCase().replace(/\s+/g, '_') || null;
  if (industryKey && INDUSTRY_RISK_MODIFIERS[industryKey]) {
    base *= INDUSTRY_RISK_MODIFIERS[industryKey];
  }
  
  return Math.min(92, Math.max(8, Math.round(base)));
```

### 6.2 Calibrate Drift Thresholds to Score Volatility
The current fixed thresholds (Job Risk: 4pts, Skill Risk: 6pts, Human Index: 8pts) are arbitrary. Calibrate to the typical score ranges:

```typescript
// Job Risk ranges 8-99 (91pt range) — 4pt threshold = 4.4% of range ✓ keep
// Skill Risk ranges 8-97 (89pt range) — 6pt threshold = 6.7% of range ✓ keep  
// Human Index ranges 0-100 (100pt range) — 8pt threshold = 8% of range ✓ keep
// BUT: also add a time-weighted significance factor:
// If drift happened within 24h: threshold × 0.75 (more sensitive)
// If drift happened > 7 days ago: threshold × 1.25 (less sensitive — fade older drift)

function getScoreDrift(): DriftResult | null {
  const history = getScoreHistory();
  if (history.length < 2) return null;
  const latest = history[history.length - 1];
  const previous = history[history.length - 2];
  const change = Math.abs(latest.plotScore - previous.plotScore);
  
  const baseThreshold = { 'job': 4, 'skill': 6, 'human-index': 8 }[latest.source] || 5;
  const ageHours = (Date.now() - previous.timestamp) / 3600000;
  const timeMultiplier = ageHours < 24 ? 0.75 : ageHours > 168 ? 1.25 : 1.0;
  const threshold = baseThreshold * timeMultiplier;
  
  if (change < threshold) return null;
  return { change, direction: latest.plotScore > previous.plotScore ? 'up' : 'down', previousDate: new Date(previous.timestamp).toLocaleDateString() };
}
```
