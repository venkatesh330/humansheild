### Future Projection — S-Curve Instead of Linear

```typescript
// v1 (WRONG — linear, doesn't saturate near limits):
yearN = baseline + N * decayFactor

// v2 (CORRECT — logistic S-curve):
function projectScore(baseline: number, decayFactor: number, years: number): number {
  const L = 97  // carrying capacity (max score)
  const k = decayFactor / 12  // growth rate
  const x0 = Math.log((L - baseline) / baseline) / k  // inflection point
  const projected = L / (1 + Math.exp(-k * (years - x0)))
  return Math.min(97, Math.max(3, Math.round(projected)))
}

// For declining scores (safe roles):
function projectSafeScore(baseline: number, years: number): number {
  // Safe roles decay risk slowly — sigmoid toward 15 floor
  return Math.max(15, Math.round(baseline - years * 1.5))
}
```

### Fix 3: Score Deduplication Logic

```typescript
// v1 (WRONG — blocks same-day improvement):

function shouldRecordScore(newScore, lastScore, lastTimestamp, threshold) {
  const scoreChanged = Math.abs(newScore - lastScore) >= threshold
  const enoughTimeElapsed = Date.now() - lastTimestamp > 60 * 60 * 1000  // 1 hour min

  // Always record if score changed meaningfully
  if (scoreChanged) return true
  // Only skip if score barely moved AND it's been less than 1 hour

  return false  // Same score, skip regardless of time
}
```
