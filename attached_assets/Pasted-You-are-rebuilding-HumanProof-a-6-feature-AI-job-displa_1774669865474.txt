2C. Fix custom skill Smart Default keyword scoring.
OLD (binary):

NEW (weighted blend):
const HUMAN_KEYWORDS = ['empathy','relationship','trust','leadership','judgment',
  'ethics','counsel','mentor','coach','presence','culture','crisis','community',
  'care','wellbeing','conflict','negotiation','persuasion','diplomacy'];
const TECH_KEYWORDS = ['code','script','automate','generate','report','analyse',
  'data','document','draft','write','design','create','process','schedule',
  'extract','classify','transcribe','summarise','calculate'];

function scoreCustomSkill(name, industry) {
  const lower = name.toLowerCase();
  let humanHits = HUMAN_KEYWORDS.filter(k => lower.includes(k)).length;
  let techHits  = TECH_KEYWORDS.filter(k => lower.includes(k)).length;
  const total = humanHits + techHits;
  let base;
  if(total === 0) base = 50;
  else if(techHits === 0) base = 10;  // pure human keywords
  else if(humanHits === 0) base = 82; // pure tech keywords
  else {
    // Blend: weighted mix
    base = Math.round((humanHits * 10 + techHits * 82) / total);
  }
  // Apply industry context
  const industryMult = INDUSTRY_MULTIPLIERS_2026[industry] || 1.0;
  return Math.min(98, Math.round(base * industryMult));
}

function shouldSaveScore(newScore: number, source: string, history: ScoreEntry[]): boolean {
  const sourceHistory = history
    .filter(e => e.source === source)
    .sort((a, b) => b.timestamp - a.timestamp);
  
  if(sourceHistory.length === 0) return true;
  
  const last = sourceHistory[0];
  const hoursSince = (Date.now() - last.timestamp) / 3600000;
  const delta = Math.abs(newScore - last.score);
  
  // Source-specific thresholds
  const thresholds: Record<string, number> = {
    'human-index': 8,   // wider tolerance — quiz variance is natural
    'job-risk':    4,   // tighter — meaningful change matters more
    'skill-risk':  6,   // medium
  };
  const threshold = thresholds[source] || 5;
  
  return hoursSince > 24 || delta > threshold;
}
